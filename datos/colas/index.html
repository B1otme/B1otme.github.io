¿Qué son las colas? #

Las colas en C++ son una estructura de datos que siguen el principio de “FIFO” (First In, First Out), lo que significa que el primer elemento que se inserta en la cola es el primero en ser eliminado.

Se utilizan metodos como el push y pop tambien para agregar o eliminar un valor de la cola.

Programa # 

#include <iostream>
#include <queue> // Utilizamos la libreria queue para las colas en c++
using namespace std;

int main(int argc, char** argv) {
	// Declaramos el nombre de la cola
	queue<int> numeros;
	int num;
	int opt;
	cout << "#Este es un ejemplo de Colas#\n";
	while(true){
	cout << "1: Insertar elemento a la cola\n2: Eliminar elemento de la cola\n3: Ver tamaño y el primer elemento de la cola\n4: Salir\n:";
	cin >> opt;
	switch (opt){
		case 1:
			cout << "Dame el numero a insertar: ";
			cin >> num;
			numeros.push(num);
			cout << "El numero fue agregado a la cola.\n";
			break;
		case 2:
			numeros.pop();
			cout << "El primer elemento de la cola fue eliminado.\n";
			break;
		case 3:
			cout << "El tamaño de la cola es: "<< numeros.size() << "\nEl numero de enfrente es: " << numeros.front() << "\n";
			break;
		case 4:
			return 0;
	}
	}
	return 0;
}
